Day 4: Cumulative Metrics & Moving Averages 
Concepts: ROWS BETWEEN (Frames), Cumulative SUM
1. Cumulative Revenue (Running Total) per Region
SELECT 
    Date,
    Region,
    Revenue,
    SUM(Revenue) OVER (PARTITION BY Region ORDER BY Date) AS Region_Running_Total
FROM 
    SalesData;

2. Moving Average (3-Sale Rolling Average) to smooth trends
SELECT 
    Date,
    Revenue,
    AVG(Revenue) OVER (
        ORDER BY Date 
        ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
    ) AS Rolling_Avg_3_Sales
FROM 
    SalesData;
