WITH SalesComparison AS (
    SELECT
        Region,
        Date,
        Product,
        Revenue,
        LAG(Revenue) OVER (PARTITION BY Region ORDER BY Date) AS Previous_Sale_Revenue
    FROM
        SalesData
)
SELECT
    Region,
    Date,
    Product,
    Revenue,
    Previous_Sale_Revenue,
    (Revenue - Previous_Sale_Revenue) AS Revenue_Change
FROM
    SalesComparison;
